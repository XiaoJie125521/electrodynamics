\newcommand{\ropepath}[1]{
	\draw[rope,black,line width=1.4pt] #1;
	\draw[rope,line width=1.1pt] #1;
}

\newcommand{\tick}[2]{
	\draw[thick] (#1) ++ (#2:0.1) --++ (#2-180:0.2);
}

\newcommand{\rightAngle}[4]{
	\pgfmathanglebetweenpoints
	{\pgfpointanchor{#2}{center}}
	{\pgfpointanchor{#3}{center}}
	\coordinate (tmpRA) at ($(#2)+(\pgfmathresult+45:#4)$);
	\draw[black]
	($(#2)!(tmpRA)!(#1)$) -- (tmpRA) -- ($(#2)!(tmpRA)!(#3)$);
}

\begin{document}
	
	\pgfmathsetmacro{\L}{3.1}
	\pgfmathsetmacro{\R}{1.0*\L}
	\pgfmathsetmacro{\Rang}{52}
	\pgfmathsetmacro{\ang}{25}
	
	\begin{tikzpicture}
		
		% ================= Coordinates =================
		
		\coordinate (O) at (0,0);
		
		\coordinate (X) at ({1.05*\L},0);
		\coordinate (Y) at (0,{1.05*\L});
		
		\coordinate (Xp) at (\ang:\L);
		\coordinate (Yp) at ({90+\ang}:\L);
		
		\coordinate (R) at (\Rang:\R);
		
		\node[fill=blue!40!black,circle,inner sep=0.9pt] (Rp) at (R) {};
		\node[above right=-1pt] at (Rp) {$P$};
		
		% Projections on fixed axes
		\coordinate (Rx) at ({\R*cos(\Rang)},0);
		\coordinate (Ry) at (0,{\R*sin(\Rang)});
		
		% Projections on rotating axes
		\coordinate (Rxp) at (\ang:{\R*cos(\Rang-\ang)});
		\coordinate (Ryp) at ({90+\ang}:{\R*sin(\Rang-\ang)});
		
		% ================= Axes =================
		
		\draw[<->,very thick,xcol!70!black]
		(X) node[right=-1pt,xcol] {$x$}
		-- (O) node[below left=-3pt] {$O$}
		-- (Y) node[left=-1pt,xcol] {$y$};
		
		\draw[<->,very thick,xcolp!70!black]
		(Xp) node[above=2pt,right=-2pt,xcolp] {$x'$}
		-- (O)
		-- (Yp) node[left=-2pt,xcolp] {$y'$};
		
		% ================= Position vector =================
		
		\draw[rvec,line cap=round]
		(O) -- (Rp)
		node[midway,left=2pt,above=1pt] {$\vb{r}$};
		
		% ================= Angle arc (plain arc, no arrow) =================
		
		\pic[
		draw,
		"$\varphi$"{scale=0.95,above=2pt,right},
		angle radius=20pt,
		angle eccentricity=1
		] {angle=X--O--Xp};
		
		% ================= Projections on fixed axes =================
		
		\begin{scope}[xcol!70!black]
			\draw[dashed] (Ry) -- (Rp) -- (Rx);
			
			\tick{Rx}{90}
			\tick{Ry}{0}
			
			\rightAngle{Rp}{Rx}{X}{0.25}
			\rightAngle{Y}{Ry}{Rp}{0.25}
		\end{scope}
		
		% ================= Projections on rotating axes =================
		
		\begin{scope}[xcolp!70!black]
			\draw[dashed] (Ryp) -- (Rp) -- (Rxp);
			
			\tick{Rxp}{90+\ang}
			\tick{Ryp}{\ang}
			
			\rightAngle{Xp}{Rxp}{Rp}{0.25}
			\rightAngle{Yp}{Ryp}{Rp}{0.25}
		\end{scope}
		
	\end{tikzpicture}
	
\end{document}
